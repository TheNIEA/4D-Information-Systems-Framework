## Quick Start for Developers

### Using the Framework Schema

```python
import json

# Load the theoretical framework
with open('framework/4d_systems_framework_schema.json', 'r') as f:
    framework = json.load(f)

# Access node definitions
nodes = framework['nodes']['definitions']
for node in nodes:
    print(f"Node {node['id']}: {node['name']} - {node['consciousnessAspect']}")

# Access sequences
sequences = framework['sequences']['primary']
for seq in sequences:
    print(f"{seq['name']}: {seq['pathNotation']} ({seq['manifestationType']})")
```

### Using the Processing Pipeline

```python
# The pipeline JSON contains executable code snippets
# Extract and run the core functions

import numpy as np

def calculate_node_development(node_id, time, alpha=0.7, beta=0.1, gamma=0.8, delta=0.05):
    """Calculate node development level over time."""
    initial_learning = alpha * np.exp(-beta * time)
    long_term_optimization = gamma * (1 - np.exp(-delta * time))
    return initial_learning + long_term_optimization

# Calculate development for Node 3 (DLPFC) at time=10
development = calculate_node_development(3, 10)
print(f"Node 3 Development at t=10: {development:.4f}")
```
